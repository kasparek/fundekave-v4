function ajax(url, vars, callbackFunction) {
  var request =  new XMLHttpRequest();
  request.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded');
  request.onreadystatechange = function() { if (request.readyState == 4 && request.status == 200) { if (request.responseText) { callbackFunction(request.responseText); } } };
  request.send(vars);
}
